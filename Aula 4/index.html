                import math

def resolver_segundo_grau(a, b, c):
    print(f"\nEquação: {a}x² + {b}x + {c} = 0")
    
    # O coeficiente 'a' não pode ser zero em uma equação do segundo grau
    if a == 0:
        return "O coeficiente 'a' deve ser diferente de 0 para ser uma equação do 2º grau."

    # Cálculo do Delta (Δ)
    delta = b**2 - 4*a*c
    print(f"Delta (Δ) = {delta}")

    if delta > 0:
        x1 = (-b + math.sqrt(delta)) / (2*a)
        x2 = (-b - math.sqrt(delta)) / (2*a)
        return f"A equação possui duas raízes reais: x1 = {x1:.2f} e x2 = {x2:.2f}"
    
    elif delta == 0:
        x = -b / (2*a)
        return f"A equação possui uma única raiz real: x = {x:.2f}"
    
    else:
        # Cálculo para raízes complexas/imaginárias
        parte_real = -b / (2*a)
        parte_imag = math.sqrt(abs(delta)) / (2*a)
        return (f"A equação possui raízes complexas: "
                f"x1 = {parte_real:.2f} + {parte_imag:.2f}i e "
                f"x2 = {parte_real:.2f} - {parte_imag:.2f}i")

# Entrada de dados pelo usuário
try:
    a = float(input("Digite o valor de a: "))
    b = float(input("Digite o valor de b: "))
    c = float(input("Digite o valor de c: "))

    resultado = resolver_segundo_grau(a, b, c)
    print(resultado)
except ValueError:
    print("Por favor, digite apenas números.")
